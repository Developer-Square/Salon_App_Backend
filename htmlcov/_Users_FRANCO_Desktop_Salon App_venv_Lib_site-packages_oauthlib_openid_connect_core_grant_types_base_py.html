49 def id_token_hash(self, value, hashfunc=hashlib.sha256): 

50 """ 

51 Its value is the base64url encoding of the left-most half of the 

52 hash of the octets of the ASCII representation of the access_token 

53 value, where the hash algorithm used is the hash algorithm used in 

54 the alg Header Parameter of the ID Token's JOSE Header. 

55 

56 For instance, if the alg is RS256, hash the access_token value 

57 with SHA-256, then take the left-most 128 bits and 

58 base64url-encode them. 

59 For instance, if the alg is HS512, hash the code value with 

60 SHA-512, then take the left-most 256 bits and base64url-encode 

61 them. The c_hash value is a case-sensitive string. 

62 

63 Example of hash from OIDC specification (bound to a JWS using RS256): 

64 

65 code: 

66 Qcb0Orv1zh30vL1MPRsbm-diHiMwcLyZvn1arpZv-Jxf_11jnpEX3Tgfvk 

67 

68 c_hash: 

69 LDktKdoQak3Pk0cnXxCltA 

70 """ 

71 digest = hashfunc(value.encode()).digest() 

72 left_most = len(digest) // 2 

73 return base64.urlsafe_b64encode(digest[:left_most]).decode().rstrip("=")

75 def add_id_token(self, token, token_handler, request, nonce=None): 

76 """ 

77 Construct an initial version of id_token, and let the 

78 request_validator sign or encrypt it. 

79 

80 The initial version can contain the fields below, accordingly 

81 to the spec: 

82 - aud 

83 - iat 

84 - nonce 

85 - at_hash 

86 - c_hash 

87 """ 

88 # Treat it as normal OAuth 2 auth code request if openid is not present 

89 if not request.scopes or 'openid' not in request.scopes: 

90 return token 

91 

92 # Only add an id token on auth/token step if asked for. 

93 if request.response_type and 'id_token' not in request.response_type: 

94 return token 

95 

96 # Implementation mint its own id_token without help. 

97 id_token = self.request_validator.get_id_token(token, token_handler, request) 

98 if id_token: 

99 token['id_token'] = id_token 

100 return token 

101 

102 # Fallback for asking some help from oauthlib framework. 

103 # Start with technicals fields bound to the specification. 

104 id_token = {} 

105 id_token['aud'] = request.client_id 

106 id_token['iat'] = int(time.time()) 

107 

108 # nonce is REQUIRED when response_type value is: 

109 # - id_token token (Implicit) 

110 # - id_token (Implicit) 

111 # - code id_token (Hybrid) 

112 # - code id_token token (Hybrid) 

113 # 

114 # nonce is OPTIONAL when response_type value is: 

115 # - code (Authorization Code) 

116 # - code token (Hybrid) 

117 if nonce is not None: 

118 id_token["nonce"] = nonce 

119 

120 # at_hash is REQUIRED when response_type value is: 

121 # - id_token token (Implicit) 

122 # - code id_token token (Hybrid) 

123 # 

124 # at_hash is OPTIONAL when: 

125 # - code (Authorization code) 

126 # - code id_token (Hybrid) 

127 # - code token (Hybrid) 

128 # 

129 # at_hash MAY NOT be used when: 

130 # - id_token (Implicit) 

131 if "access_token" in token: 

132 id_token["at_hash"] = self.id_token_hash(token["access_token"]) 

133 

134 # c_hash is REQUIRED when response_type value is: 

135 # - code id_token (Hybrid) 

136 # - code id_token token (Hybrid) 

137 # 

138 # c_hash is OPTIONAL for others. 

139 if "code" in token: 

140 id_token["c_hash"] = self.id_token_hash(token["code"]) 

141 

142 # Call request_validator to complete/sign/encrypt id_token 

143 token['id_token'] = self.request_validator.finalize_id_token(id_token, token, token_handler, request) 

144 

145 return token

147 def openid_authorization_validator(self, request): 

148 """Perform OpenID Connect specific authorization request validation. 

149 

150 nonce 

151 OPTIONAL. String value used to associate a Client session with 

152 an ID Token, and to mitigate replay attacks. The value is 

153 passed through unmodified from the Authentication Request to 

154 the ID Token. Sufficient entropy MUST be present in the nonce 

155 values used to prevent attackers from guessing values 

156 

157 display 

158 OPTIONAL. ASCII string value that specifies how the 

159 Authorization Server displays the authentication and consent 

160 user interface pages to the End-User. The defined values are: 

161 

162 page - The Authorization Server SHOULD display the 

163 authentication and consent UI consistent with a full User 

164 Agent page view. If the display parameter is not specified, 

165 this is the default display mode. 

166 

167 popup - The Authorization Server SHOULD display the 

168 authentication and consent UI consistent with a popup User 

169 Agent window. The popup User Agent window should be of an 

170 appropriate size for a login-focused dialog and should not 

171 obscure the entire window that it is popping up over. 

172 

173 touch - The Authorization Server SHOULD display the 

174 authentication and consent UI consistent with a device that 

175 leverages a touch interface. 

176 

177 wap - The Authorization Server SHOULD display the 

178 authentication and consent UI consistent with a "feature 

179 phone" type display. 

180 

181 The Authorization Server MAY also attempt to detect the 

182 capabilities of the User Agent and present an appropriate 

183 display. 

184 

185 prompt 

186 OPTIONAL. Space delimited, case sensitive list of ASCII string 

187 values that specifies whether the Authorization Server prompts 

188 the End-User for reauthentication and consent. The defined 

189 values are: 

190 

191 none - The Authorization Server MUST NOT display any 

192 authentication or consent user interface pages. An error is 

193 returned if an End-User is not already authenticated or the 

194 Client does not have pre-configured consent for the 

195 requested Claims or does not fulfill other conditions for 

196 processing the request. The error code will typically be 

197 login_required, interaction_required, or another code 

198 defined in Section 3.1.2.6. This can be used as a method to 

199 check for existing authentication and/or consent. 

200 

201 login - The Authorization Server SHOULD prompt the End-User 

202 for reauthentication. If it cannot reauthenticate the 

203 End-User, it MUST return an error, typically 

204 login_required. 

205 

206 consent - The Authorization Server SHOULD prompt the 

207 End-User for consent before returning information to the 

208 Client. If it cannot obtain consent, it MUST return an 

209 error, typically consent_required. 

210 

211 select_account - The Authorization Server SHOULD prompt the 

212 End-User to select a user account. This enables an End-User 

213 who has multiple accounts at the Authorization Server to 

214 select amongst the multiple accounts that they might have 

215 current sessions for. If it cannot obtain an account 

216 selection choice made by the End-User, it MUST return an 

217 error, typically account_selection_required. 

218 

219 The prompt parameter can be used by the Client to make sure 

220 that the End-User is still present for the current session or 

221 to bring attention to the request. If this parameter contains 

222 none with any other value, an error is returned. 

223 

224 max_age 

225 OPTIONAL. Maximum Authentication Age. Specifies the allowable 

226 elapsed time in seconds since the last time the End-User was 

227 actively authenticated by the OP. If the elapsed time is 

228 greater than this value, the OP MUST attempt to actively 

229 re-authenticate the End-User. (The max_age request parameter 

230 corresponds to the OpenID 2.0 PAPE [OpenID.PAPE] max_auth_age 

231 request parameter.) When max_age is used, the ID Token returned 

232 MUST include an auth_time Claim Value. 

233 

234 ui_locales 

235 OPTIONAL. End-User's preferred languages and scripts for the 

236 user interface, represented as a space-separated list of BCP47 

237 [RFC5646] language tag values, ordered by preference. For 

238 instance, the value "fr-CA fr en" represents a preference for 

239 French as spoken in Canada, then French (without a region 

240 designation), followed by English (without a region 

241 designation). An error SHOULD NOT result if some or all of the 

242 requested locales are not supported by the OpenID Provider. 

243 

244 id_token_hint 

245 OPTIONAL. ID Token previously issued by the Authorization 

246 Server being passed as a hint about the End-User's current or 

247 past authenticated session with the Client. If the End-User 

248 identified by the ID Token is logged in or is logged in by the 

249 request, then the Authorization Server returns a positive 

250 response; otherwise, it SHOULD return an error, such as 

251 login_required. When possible, an id_token_hint SHOULD be 

252 present when prompt=none is used and an invalid_request error 

253 MAY be returned if it is not; however, the server SHOULD 

254 respond successfully when possible, even if it is not present. 

255 The Authorization Server need not be listed as an audience of 

256 the ID Token when it is used as an id_token_hint value. If the 

257 ID Token received by the RP from the OP is encrypted, to use it 

258 as an id_token_hint, the Client MUST decrypt the signed ID 

259 Token contained within the encrypted ID Token. The Client MAY 

260 re-encrypt the signed ID token to the Authentication Server 

261 using a key that enables the server to decrypt the ID Token, 

262 and use the re-encrypted ID token as the id_token_hint value. 

263 

264 login_hint 

265 OPTIONAL. Hint to the Authorization Server about the login 

266 identifier the End-User might use to log in (if necessary). 

267 This hint can be used by an RP if it first asks the End-User 

268 for their e-mail address (or other identifier) and then wants 

269 to pass that value as a hint to the discovered authorization 

270 service. It is RECOMMENDED that the hint value match the value 

271 used for discovery. This value MAY also be a phone number in 

272 the format specified for the phone_number Claim. The use of 

273 this parameter is left to the OP's discretion. 

274

278 that the Authorization Server is being requested to use for

280 appearing in order of preference. The Authentication Context
